`authInterrupts`設定オプションを使用すると、アプリケーションで[`forbidden`](/docs/app/api-reference/functions/forbidden)と[`unauthorized`](/docs/app/api-reference/functions/unauthorized) APIを利用できます。これらの関数は実験的であるため、`next.config.js`ファイルで`authInterrupts`オプションを有効にする必要があります：

<Tabs>
<TabItem value="ts" label="TypeScript">

```ts title="next.config.ts" switcher
import type { NextConfig } from 'next'

// 実験的機能のauthInterruptsをtrueに設定します
const nextConfig: NextConfig = {
  experimental: {
    authInterrupts: true,
  },
}

export default nextConfig
```

</TabItem>
<TabItem value="js" label="JavaScript">

```js title="next.config.js" switcher
// 実験的機能のauthInterruptsをtrueに設定します
module.exports = {
  experimental: {
    authInterrupts: true,
  },
}
```

</TabItem>
</Tabs>

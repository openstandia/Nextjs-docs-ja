---
title: useLightningcss
description: Lightning CSSの実験的なサポートを有効にする。
version: experimental
---

Rustで書かれた高速なCSSバンドラー・ミニファイヤーである[Lightning CSS](https://lightningcss.dev)を使用するための実験的なサポート。

```ts title="next.config.ts" switcher
import type { NextConfig } from 'next'

const nextConfig: NextConfig = {
  experimental: {
    useLightningcss: true, // Lightning CSSを有効にする
  },
}

export default nextConfig
```

```js title="next.config.js" switcher
/** @type {import('next').NextConfig} */
const nextConfig = {
  experimental: {
    useLightningcss: true, // Lightning CSSを有効にする
  },
}

module.exports = nextConfig
```
